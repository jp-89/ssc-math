$cosecθ = 1/{sinθ} $
$cosθ = 1/{secθ} $
$secθ = 1/{cosθ} $
$tanθ = 1/{cotθ} $

$1+tan^2θ = sec^2θ$
$1+cot^2θ = cosec^2θ$
$tanθ = {sinθ}/{cosθ} $
$cotθ = {cosθ}/{sinθ} $
