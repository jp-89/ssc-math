$(a^2- b ^2)(x^2- y^2) +4abxy$
=$a ^2x^2- a^2 y ^2 -b^2 x^2 + b ^2y ^2 +4abxy$
=$a^2x^2 +2abxy +b ^2y^2 - a^2 y^2 + 2abxy - b ^2x^2$
=$(a^2x^2 +2abxy +b ^2y^2) -( a^2 y^2- 2abxy +b ^2x^2) $
=$( ax+ by)^2 - ( ay - bx) ^2$
